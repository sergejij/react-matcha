CREATE PROCEDURE GetProfilePreviewInfo (
	user_id BIGINT,

	OUT name VARCHAR(128),
	OUT surname VARCHAR(128)
)

BEGIN

	IF NOT EXISTS(SELECT * FROM user_data WHERE id = user_id)
	THEN
		SIGNAL SQLSTATE '45000' SET MYSQL_ERRNO = 404, MESSAGE_TEXT = 'Пользователь с таким ID не найден';
	END IF;

	SELECT
		user_full_data.name, user_full_data.surname 
	INTO
		name, surname 
	FROM
		user_full_data 
	WHERE
		user_full_data.id = user_id;

END
