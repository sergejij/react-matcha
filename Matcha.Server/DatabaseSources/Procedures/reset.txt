CREATE PROCEDURE RESET()

BEGIN

    SET FOREIGN_KEY_CHECKS = 0;

    TRUNCATE user_data;
    TRUNCATE user_full_data;
    TRUNCATE emails_for_confirm;
    TRUNCATE sessions;
    TRUNCATE visits;
    TRUNCATE users_interests;
    TRUNCATE likes;
    TRUNCATE dislikes;
    TRUNCATE sessions_geopositions;
    TRUNCATE messages;
    TRUNCATE chats;

    SET FOREIGN_KEY_CHECKS = 1;

END
