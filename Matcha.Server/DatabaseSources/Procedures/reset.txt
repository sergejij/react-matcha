CREATE PROCEDURE RESET()

BEGIN

    SET FOREIGN_KEY_CHECKS = 0;

    TRUNCATE user_data;
    TRUNCATE user_full_data;
    TRUNCATE emails_for_confirm;
    TRUNCATE sessions;
    TRUNCATE sessions_geolocations;
    TRUNCATE visits;
    TRUNCATE interests_list;
    TRUNCATE users_interests;
    TRUNCATE likes;

    SET FOREIGN_KEY_CHECKS = 1;

END
