CREATE FUNCTION ConfirmEmail(
    code VARCHAR(36)
) RETURNS BIGINT

BEGIN
    
    DECLARE code_bin binary(16);
    DECLARE user_id BIGINT;

    SET code_bin = UUID_TO_BIN(code);

    IF EXISTS(SELECT * FROM emails_for_confirm WHERE emails_for_confirm.confirmation_code = code_bin)
    THEN
        SET user_id = (SELECT emails_for_confirm.user_id FROM emails_for_confirm WHERE emails_for_confirm.confirmation_code = code_bin);

        UPDATE user_data
        SET email = (SELECT email FROM emails_for_confirm WHERE emails_for_confirm.user_id = user_id)
        WHERE user_data.id = user_id;

        DELETE FROM emails_for_confirm WHERE emails_for_confirm.confirmation_code = code_bin;

        RETURN user_id;
    ELSE
        SIGNAL SQLSTATE '45000' SET MYSQL_ERRNO = 422, MESSAGE_TEXT = 'Ссылка недействительна';
    END IF;

END
